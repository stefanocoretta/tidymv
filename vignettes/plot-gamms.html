Stefano Coretta

2017-10-30

set.seed(10)
data <- gamSim(4)
## Factor `by' variable example
model <- gam(
    y ~
        fac +
        s(x2, by = fac) +
        s(x0),
    data = data
)
plot_gamsd(
    model = model,
    view = "x2",
    comparison = list(fac = c("1", "2"))
)
## Summary:
##  * fac : factor; set to the value(s): 1, 2. 
##  * x2 : numeric predictor; with 100 values ranging from 0.001344 to 0.999801. 
##  * x0 : numeric predictor; set to the value(s): 0.500503229675815.

plot_gamsd(
    model = model,
    view = "x2",
    comparison = list(fac = c("1", "3")),
    bw = TRUE
)
## Summary:
##  * fac : factor; set to the value(s): 1, 3.
